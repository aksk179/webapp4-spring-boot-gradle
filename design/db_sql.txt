-- DROP TABLE BBS_CONFIG;
CREATE TABLE IF NOT EXISTS BBS_CONFIG(
	BBS_NO		INTEGER	PRIMARY KEY,
	BBS_TITLE	VARCHAR(100),
	BBS_TYPE	VARCHAR(10),
	BBS_WRITE_USE_YN	VARCHAR(1),
	BBS_REPLY_USE_YN	VARCHAR(1),
	BBS_WRITER_VIEW_YN	VARCHAR(1)
);

-- DROP TABLE BBS_BOARD;
CREATE TABLE IF NOT EXISTS BBS_BOARD(
	BBS_NO		INTEGER	NOT NULL,
	BBS_ANO		INTEGER	NOT NULL,
	BBS_TITLE	VARCHAR(100),
	BBS_WRITE_ID	VARCHAR(10),
	BBS_REGISTER_DT	TIMESTAMP,
	BBS_VIEW_CNT	INTEGER,
	BBS_GOOD_CNT	INTEGER,
	BBS_CONTENTS	TEXT,
	PRIMARY KEY(BBS_NO, BBS_ANO)
);

-- DROP TABLE BBS_COMMENT;
CREATE TABLE IF NOT EXISTS BBS_COMMENT(
	BBS_NO		INTEGER	NOT NULL,
	BBS_ANO		INTEGER	NOT NULL,
	BBS_CMT_NO	INTEGER NOT NULL,
	BBS_COMMENT	VARCHAR(1000),
	BBS_WRITER_ID	VARCHAR(10),
	BBS_REGISTER_DT	TIMESTAMP,
	PRIMARY KEY(BBS_NO, BBS_ANO, BBS_CMT_NO)
);


-- DROP TABLE BBS_ATTACH;
CREATE TABLE IF NOT EXISTS BBS_ATTACH(
	BBS_NO		INTEGER	NOT NULL,
	BBS_ANO		INTEGER	NOT NULL,
	BBS_ATTACH_FILE_NO	INTEGER	NOT NULL,
	BBS_ATTACH_FILE_NM	VARCHAR(1000),
	BBS_ATTACH_FILE_SIZE	VARCHAR(10),
	BBS_ATTACH_UPLOAD_DT	TIMESTAMP,
	BBS_WRITER_ID	VARCHAR(10),
	BBS_REGISTER_DT	TIMESTAMP,
	PRIMARY KEY(BBS_NO, BBS_ANO, BBS_ATTACH_FILE_NO)
);
